

Ingresos Totales = SUMX(JOYERIA_JPV, JOYERIA_JPV[PRECIO_VENTAS] * JOYERIA_JPV[CANTIDAD])

RANGKING = 
RANKX(
    ALL(JOYERIA_JPV[PRODUCTO]),
    [Ingresos Totales]
)


TOTAL ING ACUMULADO = 
CALCULATE(
    [Ingresos Totales], 
    TOPN(
        [RANGKING], 
        ALL(JOYERIA_JPV[PRODUCTO]), 
        [Ingresos Totales]
    )
)



% Acumulado de Ventas = 
[TOTAL ING ACUMULADO] / CALCULATE([Ingresos Totales], ALL(JOYERIA_JPV[PRODUCTO]))
